# Algolia Search for Docusaurus V3

Customizing Algolia DocSearch for your Docusaurus V3 documentation. Learn [**more**](https://docusaurus.io/docs/search#using-algolia-docsearch).

## Getting started

This configuration assumes you already have an **approved Algolia account**. Otherwise, read the [**checklist**](https://docsearch.algolia.com/docs/who-can-apply/) and [**apply for DocSearch program**](https://docsearch.algolia.com/apply/).

### Install Packages

```bash npm2yarn
npm install @docsearch/react@3
```

Alternatively, you can add the CSS to the `<head>` of your website if you don't want to use a package manager:

```js
<script src="https://cdn.jsdelivr.net/npm/@docsearch/react@3"></script>
```

### Performance optimization

Add the following on your site configuration:

```tsx title="docusaurus.config.ts"
import type { Config } from "@docusaurus/types";
import type * as Preset from "@docusaurus/preset-classic";

const config: Config = {
  title: "My Site",
  favicon: "img/favicon.ico",

  // highlight-start
  scripts: [
    {
      src: "https://YOUR_APP_ID-dsn.algolia.net",
      rel: "preconnect",
      crossorigin: true,
    },
  ],
  // highlight-end

  /* Your site config here */

  presets: [
    [
      "classic",
      {
        /* Your preset config here */
      } satisfies Preset.Options,
    ],
  ],

  themeConfig: {
    /* Your theme config here */
  } satisfies Preset.ThemeConfig,
};

export default config;
```

## Utitizing Swizzle

Start by creating a wrapper theme component using the following command:

```bash npm2yarn
npm run swizzle @docusaurus/theme-classic SearchBar --typescript
```

```tsx title="src/theme/SearchBar.tsx"
import React from "react";
import SearchBar from "@theme-original/SearchBar";
import type SearchBarType from "@theme/SearchBar";
import type { WrapperProps } from "@docusaurus/types";
import { DocSearch } from "@docsearch/react";

type Props = WrapperProps<typeof SearchBarType>;

export default function SearchBarWrapper(props: Props): JSX.Element {
  return (
    <>
      // highlight-start
      <DocSearch
        appId="YOUR-APP-ID"
        indexName="YOUR-INDEX-NAME"
        apiKey="YOUR-API-KEY"
      />
      // highlight-end
      <SearchBar {...props} />
    </>
  );
}
```
