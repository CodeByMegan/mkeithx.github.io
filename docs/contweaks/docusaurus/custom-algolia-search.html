Algolia Search for Docusaurus V3

Customizing Algolia DocSearch for your Docusaurus V3 documentation. Learn more.

Getting started​

This configuration assumes you already have an approved Algolia account. Otherwise, read the checklist and apply for DocSearch program.

Install Packages​

  • npm
  • Yarn
  • pnpm
npm install @docsearch/react@3
yarn add @docsearch/react@3
pnpm add @docsearch/react@3

Alternatively, you can add the CSS to the <head> of your website if you don't want to use a package manager:

<script src="https://cdn.jsdelivr.net/npm/@docsearch/react@3"></script>

Performance optimization​

Add the following on your site configuration:

docusaurus.config.ts
import type { Config } from "@docusaurus/types";
import type * as Preset from "@docusaurus/preset-classic";

const config: Config = {
  title: "My Site",
  favicon: "img/favicon.ico",

  scripts: [
    {
      src: "https://YOUR_APP_ID-dsn.algolia.net",
      rel: "preconnect",
      crossorigin: true,
    },
  ],

  /* Your site config here */

  presets: [
    [
      "classic",
      {
        /* Your preset config here */
      } satisfies Preset.Options,
    ],
  ],

  themeConfig: {
    /* Your theme config here */
  } satisfies Preset.ThemeConfig,
};

export default config;

Utitizing Swizzle​

Start by creating a wrapper theme component using the following command:

  • npm
  • Yarn
  • pnpm
npm run swizzle @docusaurus/theme-classic SearchBar --typescript
yarn swizzle @docusaurus/theme-classic SearchBar --typescript
pnpm run swizzle @docusaurus/theme-classic SearchBar --typescript
src/theme/SearchBar.tsx
import React from "react";
import SearchBar from "@theme-original/SearchBar";
import type SearchBarType from "@theme/SearchBar";
import type { WrapperProps } from "@docusaurus/types";
import { DocSearch } from "@docsearch/react";

type Props = WrapperProps<typeof SearchBarType>;

export default function SearchBarWrapper(props: Props): JSX.Element {
  return (
    <>
      <DocSearch
        appId="YOUR-APP-ID"
        indexName="YOUR-INDEX-NAME"
        apiKey="YOUR-API-KEY"
      />
      <SearchBar {...props} />
    </>
  );
}

Configuring Crawlers​

Algolia provides config templates for various SSG including Docusaurus to ensure optimal search experience. Feel to copy the my custom configuration:

Crawlers
Custom config

new Crawler({
  appId: 'YOUR_APP_ID',
  apiKey: 'YOUR_API_KEY',
  rateLimit: 8,
  maxDepth: 10,
  schedule: "at 10:33 on Monday",
  startUrls: ['https://YOUR_WEBSITE_URL/'],
  sitemaps: ['https://YOUR_WEBSITE_URL/sitemap.xml'],
  ignoreCanonicalTo: true,
  discoveryPatterns: ['https://YOUR_WEBSITE_URL/**'],
  actions: [
    {
      indexName: 'YOUR_INDEX_NAME',
      pathsToMatch: ['https://YOUR_WEBSITE_URL/**'],
      recordExtractor: ({ $, helpers }) => {
        // priority order: deepest active sub list header -> navbar active item -> 'Documentation'
        const lvl0 =
          $(
            '.menu__link.menu__link--sublist.menu__link--active, .navbar__item.navbar__link--active'
          )
            .last()
            .text() || 'Documentation';

        return helpers.docsearch({
          recordProps: {
            lvl0: {
              selectors: '',
              defaultValue: lvl0,
            },
            lvl1: ['header h1', 'article h1'],
            lvl2: 'article h2',
            lvl3: 'article h3',
            lvl4: 'article h4',
            lvl5: 'article h5, article td:first-child',
            lvl6: 'article h6',
            content: 'article p, article li, article td:last-child',
          },
          indexHeadings: true,
          aggregateContent: true,
          recordVersion: 'v3',
        });
      },
    },
  ],
  initialIndexSettings: {
    YOUR_INDEX_NAME: {
      attributesForFaceting: [
        'type',
        'lang',
        'language',
        'version',
        'docusaurus_tag',
      ],
      attributesToRetrieve: [
        'hierarchy',
        'content',
        'anchor',
        'url',
        'url_without_anchor',
        'type',
      ],
      attributesToHighlight: ['hierarchy', 'content'],
      attributesToSnippet: ['content:10'],
      camelCaseAttributes: ['hierarchy', 'content'],
      searchableAttributes: [
        'unordered(hierarchy.lvl0)',
        'unordered(hierarchy.lvl1)',
        'unordered(hierarchy.lvl2)',
        'unordered(hierarchy.lvl3)',
        'unordered(hierarchy.lvl4)',
        'unordered(hierarchy.lvl5)',
        'unordered(hierarchy.lvl6)',
        'content',
      ],
      distinct: true,
      attributeForDistinct: 'url',
      customRanking: [
        'desc(weight.pageRank)',
        'desc(weight.level)',
        'asc(weight.position)',
      ],
      ranking: [
        'words',
        'filters',
        'typo',
        'attribute',
        'proximity',
        'exact',
        'custom',
      ],
      highlightPreTag: '<span class="algolia-docsearch-suggestion--highlight">',
      highlightPostTag: '</span>',
      minWordSizefor1Typo: 3,
      minWordSizefor2Typos: 7,
      allowTyposOnNumericTokens: false,
      minProximity: 1,
      ignorePlurals: true,
      advancedSyntax: true,
      attributeCriteriaComputedByMinProximity: true,
      removeWordsIfNoResults: 'allOptional',
      separatorsToIndex: '_',
    },
  },
});
